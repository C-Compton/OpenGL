203 void wiic_rumble(struct wiimote_t* wm, int status) {
204  byte buf;
205 
206  if (!wm || !WIIMOTE_IS_CONNECTED(wm))
207  return;

294 int wiic_set_report_type(struct wiimote_t* wm) {
295  byte buf[2];
296  int motion, exp, ir;
297 
298  if (!wm || !WIIMOTE_IS_CONNECTED(wm))
299  return 0;
300 
301  buf[0] = (WIIMOTE_IS_FLAG_SET(wm, WIIC_CONTINUOUS) ? 0x04 : 0x00); /* set to 0x04 for continuous reporting */
302  buf[1] = 0x00;
303 
304  motion = WIIMOTE_IS_SET(wm, WIIMOTE_STATE_ACC);
305  exp = WIIMOTE_IS_SET(wm, WIIMOTE_STATE_EXP);
306  ir = WIIMOTE_IS_SET(wm, WIIMOTE_STATE_IR);
307 
308  if (motion && ir && exp) buf[1] = WM_RPT_BTN_ACC_IR_EXP;
309  else if (motion && exp) buf[1] = WM_RPT_BTN_ACC_EXP;
310  else if (motion && ir) buf[1] = WM_RPT_BTN_ACC_IR;
311  else if (ir && exp) buf[1] = WM_RPT_BTN_IR_EXP;
312  else if (ir) buf[1] = WM_RPT_BTN_ACC_IR;
313  else if (exp) buf[1] = WM_RPT_BTN_EXP;
314  else if (motion) buf[1] = WM_RPT_BTN_ACC;
315  else buf[1] = WM_RPT_BTN;
316 
317  WIIC_DEBUG("Setting report type: 0x%x", buf[1]);
318 
319  exp = wiic_send(wm, WM_CMD_REPORT_TYPE, buf, 2);
320  if (exp <= 0)
321  return exp;
322 
323  return buf[1];
324 }

447 void wiic_send_next_pending_read_request(struct wiimote_t* wm) {
448  byte buf[6];
449  struct read_req_t* req;
450 
451  if (!wm || !WIIMOTE_IS_CONNECTED(wm))
452  return;
453  if (!wm->read_req) return;
454 
455  /* skip over dirty ones since they have already been read */
456  req = wm->read_req;
457  while (req && req->dirty)
458  req = req->next;
459  if (!req)
460  return;
461 
462  /* the offset is in big endian */
463  *(unsigned int*)(buf) = BIG_ENDIAN_LONG(req->addr);
464  /* the length is in big endian */
465  *(unsigned short*)(buf + 4) = BIG_ENDIAN_SHORT(req->size);
466 
467  WIIC_DEBUG("Request read at address: 0x%x length: %i", req->addr, req->size);
468  wiic_send(wm, WM_CMD_READ_DATA, buf, 6);
469 }

520 int wiic_write_data(struct wiimote_t* wm, unsigned int addr, byte* data, byte len) {
521  byte buf[21] = {0}; /* the payload is always 23 */
522 
523  if (!wm || !WIIMOTE_IS_CONNECTED(wm))
524  return 0;
525  if (!data || !len)
526  return 0;
527 
528  WIIC_DEBUG("Writing %i bytes to memory location 0x%x...", len, addr);
529 
530  #ifdef WITH_WIIC_DEBUG
531  {
532  int i = 0;
533  printf("Write data is: ");
534  for (; i < len; ++i)
535  printf("%x ", data[i]);
536  printf("\n");
537  }
538  #endif
539 
540  /* the offset is in big endian */
541  *(int*)(buf) = BIG_ENDIAN_LONG(addr);
542 
543  /* length */
544  *(byte*)(buf + 4) = len;
545 
546  /* data */
547  memcpy(buf + 5, data, len);
548 
549  wiic_send(wm, WM_CMD_WRITE_DATA, buf, 21);
550  return 1;
551 }

564 int wiic_send(struct wiimote_t* wm, byte report_type, byte* msg, int len) {
565  byte buf[32]; /* no payload is better than this */
566  int rumble = 0;
567 
568  buf[0] = WM_SET_REPORT | WM_BT_OUTPUT;
569  buf[1] = report_type;
570 
571  switch (report_type) {
572  case WM_CMD_LED:
573  case WM_CMD_RUMBLE:
574  case WM_CMD_CTRL_STATUS:
575  case WM_CMD_REPORT_TYPE:
576  {
577  /* Rumble flag for: 0x11, 0x13, 0x14, 0x15, 0x19 or 0x1a */
578  if (WIIMOTE_IS_SET(wm, WIIMOTE_STATE_RUMBLE))
579  rumble = 1;
580  break;
581  }
582  default:
583  break;
584  }
585 
586  memcpy(buf+2, msg, len);
587  if (rumble)
588  buf[2] |= 0x01;
589  else
590  buf[2] &= 0xFE; // Fix for Wiiuse bug
591 
592  #ifdef WITH_WIIC_DEBUG
593  {
594  int x = 2;
595  printf("[DEBUG] (id %i) SEND: (%x) %.2x ", wm->unid, buf[0], buf[1]);
596  for (; x < len+2; ++x)
597  printf("%.2x ", buf[x]);
598  printf("\n");
599  }
600  #endif
601 
602  return wiic_io_write(wm, buf, len+2);
603 }
