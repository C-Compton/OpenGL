61 void landGen( std::vector<point4> &vec, std::vector<unsigned int> &drawIndex ) {
62 
63  Timer Tick;
64 
65  const int _N = 5;
66  const int S = pow(2,_N) + 1;
67 
68  if (DEBUG) printf( "\nEntering landGen()...\n");
69  // the range (-h -> h) for the average offset
70  // This determines the jaggedness of the peaks and valleys.
71  // A smaller initial value will create smaller peaks and shollow valleys for
72  // a look more similar to rolling hill tops.
73  double h = 5.0;
74  double CH = 0.5;
75  double magnitude = (h*(2 - pow(2,-(_N))));
76  fprintf( stderr, "landGen theoretical magnitude: %f\n", magnitude );
77 
78  /* Initialize all points in the vector to have their X,Z (and w) coordinates. */
79  if (vec.size()) vec.clear();
80  vec.reserve( S * S );
81  for ( int i = 0; i < S; ++i )
82  for ( int j = 0; j < S; ++j )
83  vec.push_back( vec4( j, 0, i, 1 ) );
84 
85  /* Initialize our color vectors. */
86  std::vector< vec4 > &col = terrain->colors;
87  if (col.size()) col.clear();
88  col.reserve( S*S );
89  for ( int i = 0; i < S; ++i )
90  for ( int j = 0; j < S; ++j )
91  col.push_back( vec4( 0.5, 0.5, 0.5, 1.0 ) );
92 
93  /* Simulate a 2D array in this 1D array. Use these Macros to help. */
94 #define OffsetAt(X,Z) ((X)*S+(Z))
95 #define VertexAt(X,Z) (vec.at(OffsetAt(X,Z)))
96 #define HeightAt(X,Z) (VertexAt(X,Z).y)
97 #define ColorAt(X,Z) (col.at(OffsetAt(X,Z)))
98 
99  // Initialize the corners of the grid
100  HeightAt( 0, 0 ) = 0;
101  HeightAt( S-1, 0 ) = 0;
102  HeightAt( 0, S-1 ) = 0;
103  HeightAt( S-1, S-1 ) = 0;
104 
105  // Populate the (x, y, z) values of vec4 according to the
106  // Diamond-Square algorithm
107  // sideLength is the distance of a single square side or
108  // distance of diagonal in diamond.
109  if (DEBUG) printf("\nEntering for( sideLength...) ...\n");
110  for (int sideLength = S-1; sideLength >= 2; sideLength /= 2, h /= 2.0, CH /= 2.0) {
111  int halfSide = sideLength / 2;
112  // generate new square values
113  for ( int x = 0 ; x < S-1 ; x += sideLength ) {
114  for ( int z = 0 ; z < S-1 ; z += sideLength) {
115  double avg = \
116  (HeightAt( x, z ) + HeightAt( x + sideLength, z ) +
117  HeightAt( x, z + sideLength ) + HeightAt( x + sideLength, z + sideLength ))/4.0;
118 
119  vec4 color_avg = \
120  (ColorAt(x,z) + ColorAt(x+sideLength,z) +
121  ColorAt(x,z+sideLength) + ColorAt(x+sideLength,z+sideLength))/4.0;
122  vec4 color_jitter = vec4( jitter(CH), jitter(CH), jitter(CH), 0 );
123 
124  HeightAt( x + halfSide, z + halfSide ) = avg + jitter(h);
125  ColorAt( x + halfSide, z + halfSide ) = color_avg + color_jitter;
126  } // for z
127  } // for x
128 
129  // Generate the diamond values
130  // Since diamonds are staggered, we only move x by half side
131  // NOTE: If the data shouldn't wrap the x < SIZE and y < SIZE
132  for ( int x = 0 ; x < S-1 ; x += halfSide ) {
133  for (int z = ( x+halfSide ) % sideLength ; z < S-1 ; z += sideLength ) {
134 
135  // x,z is center of diamond
136  // We must use mod and add SIZE for subtraction
137  // so that we can wrap around the array to find the corners
138 
139  double avg =
140  (HeightAt( (x-halfSide + S) % S, z ) +
141  HeightAt( x + halfSide % S, z ) +
142  HeightAt( x, z + halfSide % S ) +
143  HeightAt( x, (z - halfSide + S) % S )) / 4.0;
144 
145  vec4 color_avg =
146  (ColorAt( (x-halfSide + S) % S, z ) +
147  ColorAt( x + halfSide % S, z ) +
148  ColorAt( x, z + halfSide % S ) +
149  ColorAt( x, (z - halfSide + S) % S )) / 4.0;
150  vec4 color_jitter = vec4( jitter(CH), jitter(CH), jitter(CH), 0 );
151 
152  HeightAt( x, z ) = avg + jitter(h);
153  ColorAt( x, z ) = color_avg + color_jitter;
154 
155  // Wrapping:
156  if (x == 0) {
157  HeightAt( S-1, z ) = HeightAt( x, z );
158  ColorAt( S-1, z ) = ColorAt( x, z );
159  }
160  if (z == 0) {
161  HeightAt( x, S-1 ) = HeightAt( x, z );
162  ColorAt( x, S-1 ) = ColorAt( x, z );
163  }
164  } // for z
165  } // for x
166  } // for sideLength
167 
168 
169  //vec.reserve(2*S*S - 2*S);
170  // Convert 2D array into 1D array to pass to shaders
171  // Size of 1D array is 2*SIZE^2 - 2SIZE for
172  // the sake of proper parsing by glTriangleStrips
173  for ( int i = 0, j = 0 ; i + 1 < S ; i++ ) {
174  for ( j = 0 ; j < S ; j++ ) {
175  drawIndex.push_back(OffsetAt(i,j));
176  drawIndex.push_back(OffsetAt(i+1,j));
177  }
178 
179  /* If we're out of rows to serialize, give up. */
180  if (++i == S) break;
181 
182  for ( ; j > 0 ; j-- ){
183  drawIndex.push_back(OffsetAt(i,j-1));
184  drawIndex.push_back(OffsetAt(i+1,j-1));
185  }
186  }
187 
188  for ( int i = 0; i < S*S; i++ ) {
189  terrain->colors.push_back(vec4(rand_float(),rand_float(),rand_float(),1.0));
190  }
191 
192  if (DEBUG) printf("\nExiting landGen()...\n");
193 
194  Tick.Tock();
195  fprintf( stderr, "Landgen took %lu usec, %f msec, %f sec to generate %d vertices.\n",
196  Tick.Delta(), Tick.Delta()/1000.0, Tick.Delta()/1000000.0, S*S );
197 
198  return;
199 }
