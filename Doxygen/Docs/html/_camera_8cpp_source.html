337 void Camera::Accel( const vec2 &raw_accel ) {
338 
339  //Accel comes in as a vector with a length between -sqrt(2) and sqrt(2).
340  //We change it to be between -MAX_ACCEL and MAX_ACCEL.
341  vec2 accel = raw_accel * (MaxAccel/SQRT2);
342 
343  //Now, we scale our accel vector so that we accelerate less when we are near MaxSpeed.
344  accel *= (1-POW5(speed_cap));
345 
346  //The acceleration is finally applied to the velocity vector.
347  velocity.x += accel.x;
348  velocity.y += accel.y;
349 
350  //speed and speed_cap must now be recalculated.
351  speed_cap = (speed = length(velocity))/MaxSpeed;
352 
353  if (0) fprintf( stderr, "Velocity: (%f,%f)\n", velocity.x, velocity.y );
354 }

383 void Camera::Idle( void ) {
384 
385  if (Motion[Camera::Forward]) surge( Camera::initSpeed );
386  else if (Motion[Camera::Backward]) surge( -Camera::initSpeed );
387 
388  if (Motion[Camera::Right]) sway( Camera::initSpeed );
389  else if (Motion[Camera::Left]) sway( -Camera::initSpeed );
390 
391  if (Motion[Camera::Up]) heave( Camera::initSpeed );
392  else if (Motion[Camera::Down]) heave( -Camera::initSpeed );
393 
394  /* Apply the Balance Board vectors, along with a correction. */
395  surge( velocity.x / 20000 );
396  sway( velocity.y / 20000 );
397  heave( velocity.z / 20000 );
398 
399  // We can only apply friction if we are moving.
400  if (speed) {
401 #define FRICTION 2
402  // Friction is a vector that is the opposite of velocity.
403  vec3 friction = -velocity;
404  // By dividing friction by (speed/FRICTION), we guarantee that the magnitude is FRICTION.
405  friction = friction / (speed/FRICTION);
406  velocity += friction;
407  speed = length(velocity);
408  speed_cap = speed/MaxSpeed;
409  }

508 void Camera::send( const glsl_var &which ) {
509 
510  switch (which) {
511  case TRANSLATION:
512  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, T );
513  send( CTM );
514  break;
515  case ROTATION:
516  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, R );
517  send( CTM );
518  break;
519  case VIEW:
520  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, P );
521  send( CTM );
522  break;
523  case CTM:
524 #ifdef POSTMULT
525  ctm = T*R*P;
526 #else
527  ctm = P*R*T;
528 #endif
529  glUniformMatrix4fv( glsl_handles[which], 1, GL_TRUE, ctm );
530  break;
531  default:
532  throw std::invalid_argument( "Unknown GLSL variable handle." );
533  }
534 }
