450 void Camera::send( const glsl_var &which ) {
451 
452  switch (which) {
453  case TRANSLATION:
454  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, T );
455  send( PRT_M );
456  break;
457  case ROTATION:
458  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, R );
459  send( PRT_M );
460  break;
461  case PERSPECTIVE:
462  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, P );
463  send( PRT_M );
464  break;
465  case TRP_M:
466  TRP = T*R*P;
467  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, TRP );
468  break;
469  case PRT_M:
470  PRT = P*R*T;
471  glUniformMatrix4fv( glsl_handles[which], 1, GL_TRUE, PRT );
472  break;
473  case CTM:
474  glUniformMatrix4fv( glsl_handles[which], 1, GL_FALSE, ctm );
475  break;
476  default:
477  throw std::invalid_argument( "Unknown GLSL variable handle." );
478  }
479 }
