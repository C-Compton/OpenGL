275 void Camera::adjustRotation( const mat4 &adjustment, const bool &fixed ) {
276 
277  // By default, the 'order' bool represents the POSTMULT behavior.
278  bool order = POSTMULT;
279  // However, if the fixed bool is present, flip it around.
280  // This stealthily inserts this rotation "first",
281  // So that it is applied before any other rotations.
282 
283  if (fixed) order = !order;
284 
285  if (order) {
286  // In a post-mult system, the argument order is left-to-right,
287  // So the adjustment appears last.
288  R = R * adjustment;
289  } else {
290  // In a pre-mult system, the last argument is applied first,
291  // So the adjustment should appear first.
292  R = adjustment * R;
293  }
294 
295  // The model needs to apply the OPPOSITE rotations, so Transpose AGAIN!
296  // We want to rotate only the object, not his position in space,
297  // So use the 'Rotate' object, not the 'Orbit'.
298  trans.rotation.Adjust( transpose(adjustment), fixed );
299  trans.CalcCTM();
300 
301  Send( ROTATION );
302 }

411 void Camera::Accel( const vec3 &raw_accel ) {
412 
413  /*
414  This scale factor is the cumulation of several scaling factors.
415  (A) (MaxAccel/SQRT3)
416  scales the vector so that it has a magnitude
417  Between -MAX_ACCEL and MAX_ACCEL.
418 
419  (B) (1-POW5(speed_cap))
420  scales the vector with relation to the
421  current speed. This effectively models "Drag".
422  The closer we are to maximum speed, this scaling factor
423  Will approach 0, so that if we are at MAX SPEED,
424  our acceleration will be scaled to (nearly) zero.
425 
426  (C) (Tick.Delta() / KeyFrameRate)
427  This scales the amount of acceleration by an amount [0,1]
428  so that if we are drawing many frames, we'll apply less acceleration.
429  If we are drawing not so many, we'll apply more.
430  This should keep the animation relatively consistent across platforms.
431  */
432 
433  float Scale = (MaxAccel/SQRT3) * (1-POW5(speed_cap)) * (Tick.Scale());
434  vec3 accel = raw_accel * Scale;
435 
436  if (DEBUG_MOTION) {
437 
438  fprintf( stderr,
439  "Accel(); raw_accel = (%f,%f,%f)\n",
440  raw_accel.x, raw_accel.y, raw_accel.z );
441  fprintf( stderr,
442  "Accel(); Scale = (MaxAccel/SQRT3) * (1-POW5(speed_cap)) * Tick.Scale()\n" );
443  fprintf( stderr,
444  "Accel(); Scale = (%f/%f) * (%f) * (%f)\n",
445  MaxAccel, SQRT3, (1-POW5(speed_cap)), Tick.Scale() );
446  fprintf( stderr,
447  "Accel(); Scale = %f\n", Scale );
448  fprintf( stderr,
449  "Accle(); accel = raw_accel * Scale = (%f,%f,%f)\n",
450  accel.x, accel.y, accel.z );
451  }
452 
453  //The acceleration is finally applied to the velocity vector.
454  velocity += accel;
455 
456  //speed and speed_cap must now be recalculated.
457  speed_cap = (speed = length(velocity))/MaxSpeed;
458 
459  if (DEBUG_MOTION)
460  fprintf( stderr, "Applied Acceleration to Velocity, Is now: (%f,%f,%f)\n",
461  velocity.x, velocity.y, velocity.z );
462 }

491 void Camera::Idle( void ) {
492 
493  /* Apply the automated motion instructions, if any --
494  These are primarily from the keyboard. */
495  if (Motion[Camera::Forward]) Accel(vec3(0,0,1));
496  if (Motion[Camera::Backward]) Accel(vec3(0,0,-1));
497  if (Motion[Camera::Right]) Accel(vec3(0,1,0));
498  if (Motion[Camera::Left]) Accel(vec3(0,-1,0));
499  if (Motion[Camera::Up]) Accel(vec3(1,0,0));
500  if (Motion[Camera::Down]) Accel(vec3(-1,0,0));
501 
502 
503  if (speed) {
504  /* Apply the velocity vectors computed from Accel,
505  which includes instructions from keyboard and the Balance Board. */
506  /* 1/20000 is a magic constant which converts our velocity units
507  into model units. */
508  /* Tick.Delta()/KeyFrameRate helps keep animation speed consistent
509  between different hardware. */
510  float UnitScale = (1.0/20000.0);
511  float Scale = Tick.Scale() * UnitScale;
512 
513  if (DEBUG_MOTION)
514  fprintf( stderr, "Applying Translation: + (%f,%f,%f)\n",
515  velocity.x * Scale, velocity.y * Scale,
516  velocity.z * Scale );
517 
518  heave( velocity.x * Scale );
519  sway( velocity.y * Scale );
520  surge( velocity.z * Scale );
521 
522 
523 
524  // Friction Calculations
525  if (speed < (FrictionMagnitude * Tick.Scale())) {
526  if (DEBUG_MOTION)
527  fprintf( stderr, "Friction has stopped all movement.\n" );
528  velocity = vec3(0,0,0);
529  speed = 0;
530  speed_cap = 0;
531  } else {
532  // Friction is a vector that is the opposite of velocity.
533  vec3 frictionVec = -velocity;
534  /* By dividing friction by (speed/FrictionMagnitude),
535  we guarantee that the magnitude is FrictionMagnitude. */
536  frictionVec = frictionVec / (speed/FrictionMagnitude);
537  frictionVec *= Tick.Scale();
538 
539  if (DEBUG_MOTION)
540  fprintf( stderr, "Applying friction to Velocity: + (%f,%f,%f)\n",
541  frictionVec.x, frictionVec.y, frictionVec.z );
542  velocity += frictionVec;
543  speed = length(velocity);
544  speed_cap = speed/MaxSpeed;
545  }
546  }
547 
548 }
