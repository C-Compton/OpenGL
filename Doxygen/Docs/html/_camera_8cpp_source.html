243 void Camera::adjustRotation( const mat4 &adjustment, const bool &fixed ) {
244 
245  // By default, the 'order' bool represents the POSTMULT behavior.
246  bool order = POSTMULT;
247  // However, if the fixed bool is present, flip it around.
248  // This stealthily inserts this rotation "first",
249  // So that it is applied before any other rotations.
250 
251  if (fixed) order = !order;
252 
253  if (order) {
254  // In a post-mult system, the argument order is left-to-right,
255  // So the adjustment appears last.
256  R = R * adjustment;
257  } else {
258  // In a pre-mult system, the last argument is applied first,
259  // So the adjustment should appear first.
260  R = adjustment * R;
261  }
262 
263  send( ROTATION );
264 
265 }

374 void Camera::Accel( const vec3 &raw_accel ) {
375 
376  /*
377  This scale factor is the cumulation of several scaling factors.
378  (A) (MaxAccel/SQRT3)
379  scales the vector so that it has a magnitude
380  Between -MAX_ACCEL and MAX_ACCEL.
381 
382  (B) (1-POW5(speed_cap))
383  scales the vector with relation to the
384  current speed. This effectively models "Drag".
385  The closer we are to maximum speed, this scaling factor
386  Will approach 0, so that if we are at MAX SPEED,
387  our acceleration will be scaled to (nearly) zero.
388 
389  (C) (Tick.Delta() / KeyFrameRate)
390  This scales the amount of acceleration by an amount [0,1]
391  so that if we are drawing many frames, we'll apply less acceleration.
392  If we are drawing not so many, we'll apply more.
393  This should keep the animation relatively consistent across platforms.
394  */
395 
396  float Scale = (MaxAccel/SQRT3) * (1-POW5(speed_cap)) * (Tick.Delta() / KeyFrameRate);
397  vec3 accel = raw_accel * Scale;
398 
399  //The acceleration is finally applied to the velocity vector.
400  velocity += accel;
401 
402  //speed and speed_cap must now be recalculated.
403  speed_cap = (speed = length(velocity))/MaxSpeed;
404 
405  if (0) fprintf( stderr, "Velocity: (%f,%f,%f)\n", velocity.x, velocity.y, velocity.z );
406 }

435 void Camera::Idle( void ) {
436 
437  /* Apply the automated motion instructions, if any --
438  These are primarily from the keyboard. */
439  if (Motion[Camera::Forward]) Accel(vec3(0,0,1));
440  if (Motion[Camera::Backward]) Accel(vec3(0,0,-1));
441  if (Motion[Camera::Right]) Accel(vec3(0,1,0));
442  if (Motion[Camera::Left]) Accel(vec3(0,-1,0));
443  if (Motion[Camera::Up]) Accel(vec3(1,0,0));
444  if (Motion[Camera::Down]) Accel(vec3(-1,0,0));
445 
446  /* Apply the velocity vectors computed from Accel,
447  which includes instructions from keyboard and the Balance Board. */
448  /* 1/20000 is a magic constant which converts our velocity units
449  into model units. */
450  /* Tick.Delta()/KeyFrameRate helps keep animation speed consistent
451  between different hardware. */
452  float TimeScale = (Tick.Delta()/KeyFrameRate);
453  float UnitScale = (1.0/20000.0);
454  float Scale = TimeScale * UnitScale;
455 
456  heave( velocity.x * Scale );
457  sway( velocity.y * Scale );
458  surge( velocity.z * Scale );
459 
460  // We can only apply friction if we are moving.
461  if (speed) {
462  // Friction is a vector that is the opposite of velocity.
463  vec3 frictionVec = -velocity;
464  /* By dividing friction by (speed/FrictionMagnitude),
465  we guarantee that the magnitude is FrictionMagnitude. */
466  frictionVec = frictionVec / (speed/FrictionMagnitude);
467  velocity += (frictionVec * TimeScale);
468  speed = length(velocity);
469  speed_cap = speed/MaxSpeed;
470 
471  }
472 }
