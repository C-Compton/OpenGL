00232 void Camera::adjustRotation( const mat4 &adjustment, const bool &fixed ) {
00233 
00234   // By default, the 'order' bool represents the POSTMULT behavior.
00235   bool order = POSTMULT;
00236   // However, if the fixed bool is present, flip it around.
00237   // This stealthily inserts this rotation "first",
00238   // So that it is applied before any other rotations.
00239 
00240   if (fixed) order = !order;
00241 
00242   if (order) {
00243     // In a post-mult system, the argument order is left-to-right,
00244     // So the adjustment appears last.
00245     R = R * adjustment;
00246   } else {
00247     // In a pre-mult system, the last argument is applied first,
00248     // So the adjustment should appear first.    
00249     R = adjustment * R;
00250   }
00251 
00252   send( ROTATION );
00253 
00254 }

00313 void Camera::pitch( const float &by, const bool &fixed ) {
00314   /*
00315     Since negative values are interpreted as pitching down,

00363 void Camera::Accel( const vec3 &raw_accel ) {
00364 
00365   //Accel comes in as a vector with a length between -sqrt(3) and sqrt(3).
00366   //We change it to be between -MAX_ACCEL and MAX_ACCEL.
00367   vec3 accel = raw_accel * (MaxAccel/SQRT3);
00368 
00369   //Now, we scale our accel vector so that we accelerate less when we are near MaxSpeed.
00370   accel *= (1-POW5(speed_cap));
00371   
00372   //The acceleration is finally applied to the velocity vector.
00373   velocity += accel;
00374 
00375   //speed and speed_cap must now be recalculated.
00376   speed_cap = (speed = length(velocity))/MaxSpeed;
00377 
00378   if (0) fprintf( stderr, "Velocity: (%f,%f,%f)\n", velocity.x, velocity.y, velocity.z );
00379 }

00408 void Camera::Idle( void ) {
00409 
00410   /* Apply the automated motion instructions, if any --
00411      These are primarily from the keyboard. */
00412   if (Motion[Camera::Forward]) Accel(vec3(0,0,1));
00413   if (Motion[Camera::Backward]) Accel(vec3(0,0,-1));
00414   if (Motion[Camera::Right]) Accel(vec3(0,1,0));
00415   if (Motion[Camera::Left]) Accel(vec3(0,-1,0));
00416   if (Motion[Camera::Up]) Accel(vec3(1,0,0));
00417   if (Motion[Camera::Down]) Accel(vec3(-1,0,0));
00418 
00419   /* Apply the velocity vectors computed from Accel,
00420      which includes instructions from keyboard and the Balance Board. */
00421 
00422   heave( velocity.x / 20000 );
00423   sway( velocity.y / 20000 );
00424   surge( velocity.z / 20000 );
00425 
00426   // We can only apply friction if we are moving. 
00427   if (speed) {
00428     // Friction is a vector that is the opposite of velocity.
00429     vec3 frictionVec = -velocity;
00430     /* By dividing friction by (speed/FrictionMagnitude), 
00431        we guarantee that the magnitude is FrictionMagnitude. */
00432     frictionVec = frictionVec / (speed/FrictionMagnitude);
00433     velocity += frictionVec;
00434     speed = length(velocity);
00435     speed_cap = speed/MaxSpeed;
00436   }
00437 }
