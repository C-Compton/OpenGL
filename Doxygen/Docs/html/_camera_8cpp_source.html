00268 void Camera::pitch( const float &by ) {
00269   /*
00270     Since negative values are interpreted as pitching down,
00271     We leave the input uninverted, because a negative rotation
00272     about the X axis rotates counter-clockwise (looking right),
00273     and clockwise (looking left), which achieves the effect of
00274     looking 'down'.
00275   */ 
00276   adjustRotation(RotateX(-by));
