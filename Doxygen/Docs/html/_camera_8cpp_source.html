242 void Camera::adjustRotation( const mat4 &adjustment, const bool &fixed ) {
243 
244  // By default, the 'order' bool represents the POSTMULT behavior.
245  bool order = POSTMULT;
246  // However, if the fixed bool is present, flip it around.
247  // This stealthily inserts this rotation "first",
248  // So that it is applied before any other rotations.
249 
250  if (fixed) order = !order;
251 
252  if (order) {
253  // In a post-mult system, the argument order is left-to-right,
254  // So the adjustment appears last.
255  R = R * adjustment;
256  } else {
257  // In a pre-mult system, the last argument is applied first,
258  // So the adjustment should appear first.
259  R = adjustment * R;
260  }
261 
262  send( ROTATION );
263 
264 }

373 void Camera::Accel( const vec3 &raw_accel ) {
374 
375  //Accel comes in as a vector with a length between -sqrt(3) and sqrt(3).
376  //We change it to be between -MAX_ACCEL and MAX_ACCEL.
377  vec3 accel = raw_accel * (MaxAccel/SQRT3);
378 
379  //Now, we scale our accel vector so that we accelerate less when we are near MaxSpeed.
380  accel *= (1-POW5(speed_cap));
381 
382  //The acceleration is finally applied to the velocity vector.
383  velocity += accel;
384 
385  //speed and speed_cap must now be recalculated.
386  speed_cap = (speed = length(velocity))/MaxSpeed;
387 
388  if (0) fprintf( stderr, "Velocity: (%f,%f,%f)\n", velocity.x, velocity.y, velocity.z );
389 }

418 void Camera::Idle( void ) {
419 
420  /* Apply the automated motion instructions, if any --
421  These are primarily from the keyboard. */
422  if (Motion[Camera::Forward]) Accel(vec3(0,0,1));
423  if (Motion[Camera::Backward]) Accel(vec3(0,0,-1));
424  if (Motion[Camera::Right]) Accel(vec3(0,1,0));
425  if (Motion[Camera::Left]) Accel(vec3(0,-1,0));
426  if (Motion[Camera::Up]) Accel(vec3(1,0,0));
427  if (Motion[Camera::Down]) Accel(vec3(-1,0,0));
428 
429  /* Apply the velocity vectors computed from Accel,
430  which includes instructions from keyboard and the Balance Board. */
431 
432  heave( velocity.x / 20000 );
433  sway( velocity.y / 20000 );
434  surge( velocity.z / 20000 );
435 
436  // We can only apply friction if we are moving.
437  if (speed) {
438  // Friction is a vector that is the opposite of velocity.
439  vec3 frictionVec = -velocity;
440  /* By dividing friction by (speed/FrictionMagnitude),
441  we guarantee that the magnitude is FrictionMagnitude. */
442  frictionVec = frictionVec / (speed/FrictionMagnitude);
443  velocity += frictionVec;
444  speed = length(velocity);
445  speed_cap = speed/MaxSpeed;
446  }
447 }
