203 void MLData::computeDisplacement(const Training* t, vector<double>& features)
204 {
205  double xspace, yspace, zspace;
206  double xvel, yvel, zvel;
207  xspace = yspace = zspace = xvel = yvel = zvel = 0.0;
208 
209  // Compute displacement and update speed
210  for(int i = 0 ; i < t->size() ; i++) {
211  const AccSample* sample = reinterpret_cast<const AccSample*>(t->sampleAt(i));
212  xspace = xspace + xvel*timeDelta_ + 0.5*sample->x()*timeDelta_*timeDelta_*GRAV_ACC;
213  xvel = xvel + sample->x()*timeDelta_*GRAV_ACC;
214  yspace = yspace + yvel*timeDelta_ + 0.5*sample->y()*timeDelta_*timeDelta_*GRAV_ACC;
215  yvel = yvel + sample->y()*timeDelta_*GRAV_ACC;
216  zspace = zspace + zvel*timeDelta_ + 0.5*(sample->z()-1)*timeDelta_*timeDelta_*GRAV_ACC;
217  zvel = zvel + (sample->z()-1)*timeDelta_*GRAV_ACC;
218  }
219 
220  features.push_back(xspace);
221  features.push_back(yspace);
