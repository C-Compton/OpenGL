261  color4 ambient_product, diffuse_product, specular_product ;

263  float dim = 1.0;

265  switch(lightMode){

267 
268  case 0: // on, no real effect
269 
270  ambient_product = light_ambient ;
271  diffuse_product = light_diffuse ;
272  specular_product = light_specular ;
273 
274  break;
275 
276 
277 
278  case 1: // strobe/blink
279 
280 
281  if( (frameNumber/30) < 3 ){ // if( (frameNumber/60) < 3 ){
282 
283  ambient_product = dark ;
284  diffuse_product = dark ;
285  specular_product = dark ;
286 
287  } else {
288 
289  ambient_product = light_ambient ;
290  diffuse_product = light_diffuse ;
291  specular_product = light_specular ;
292 
293  }
294 
295  break;
296 
297 
298  case 2: // pulse
299 
300  dim = sin( pulseRad * frameNumber) ;
301  dim *= dim ; // sin squared for great justice and positivity
302 
303  ambient_product = light_ambient * dim;
304  diffuse_product = light_diffuse * dim;
305  specular_product = light_specular * dim;
306 
307  break;
308

312  ambient_product = dark ;
313  diffuse_product = dark ;
314  specular_product = dark ;
315 
316  break;
