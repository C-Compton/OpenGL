00040 int balance_board_handshake(struct wiimote_t* wm, struct balance_board_t* bb, byte* data, unsigned short len) 
00041 {
00042     int offset = 4;
00043     
00044     bb->cal_low_weight.top_left = 0;
00045     bb->cal_low_weight.top_right = 0;   
00046     bb->cal_low_weight.bottom_left = 0;
00047     bb->cal_low_weight.bottom_right = 0;
00048     bb->cal_medium_weight.top_left = 0;
00049     bb->cal_medium_weight.top_right = 0;    
00050     bb->cal_medium_weight.bottom_left = 0;
00051     bb->cal_medium_weight.bottom_right = 0;
00052     bb->cal_high_weight.top_left = 0;
00053     bb->cal_high_weight.top_right = 0;  
00054     bb->cal_high_weight.bottom_left = 0;
00055     bb->cal_high_weight.bottom_right = 0;
00056     bb->pressure_raw_data.top_left = 0;
00057     bb->pressure_raw_data.top_right = 0;    
00058     bb->pressure_raw_data.bottom_left = 0;
00059     bb->pressure_raw_data.bottom_right = 0; 
00060     bb->pressure_weight.top_left = 0.0;
00061     bb->pressure_weight.top_right = 0.0;    
00062     bb->pressure_weight.bottom_left = 0.0;
00063     bb->pressure_weight.bottom_right = 0.0; 
00064     bb->pressure_weight.weight = 0.0;
00065     
00066     /*
00067      *  Sometimes the data returned here is not correct (all bytes are 0xFF).
00068      *  This might happen because the wiimote is lagging
00069      *  behind our initialization sequence.
00070      *  To fix this just request the handshake again.    
00071      */
00072     if (data[offset+offset] == 0xFF && data[offset+offset+16] == 0xFF) {
00073         /* get the calibration data */
00074         byte* handshake_buf = malloc(EXP_HANDSHAKE_LEN * sizeof(byte));
00075 
00076         WIIC_DEBUG("Balance board handshake appears invalid, trying again.");
00077         wiic_read_data_cb(wm, handshake_expansion, handshake_buf, WM_EXP_MEM_CALIBR, EXP_HANDSHAKE_LEN);
00078 
00079         return 0;
00080     }
00081     
00082     bb->cal_low_weight.top_right = (data[offset+0] << 8) | data[offset+1] ;
00083     bb->cal_low_weight.bottom_right = (data[offset+2] << 8) | data[offset+3] ;
00084     bb->cal_low_weight.top_left = (data[offset+4] << 8) | data[offset+5] ;
00085     bb->cal_low_weight.bottom_left = (data[offset+6] << 8) | data[offset+7] ;
00086     bb->cal_medium_weight.top_right = (data[offset+8] << 8) | data[offset+9] ;
00087     bb->cal_medium_weight.bottom_right = (data[offset+10] << 8) | data[offset+11] ;
00088     bb->cal_medium_weight.top_left = (data[offset+12] << 8) | data[offset+13] ;
00089     bb->cal_medium_weight.bottom_left = (data[offset+14] << 8) | data[offset+15] ;
00090     bb->cal_high_weight.top_right = (data[offset+16] << 8) | data[offset+17] ;
00091     bb->cal_high_weight.bottom_right = (data[offset+18] << 8) | data[offset+19] ;
00092     bb->cal_high_weight.top_left = (data[offset+20] << 8) | data[offset+21] ;
00093     bb->cal_high_weight.bottom_left = (data[offset+22] << 8) | data[offset+23] ;
00094 
00095     /* handshake done */
00096     wm->exp.type = EXP_BALANCE_BOARD;
00097 
00098     return 1;
00099 }

00107 void balance_board_event(struct balance_board_t* bb, byte* msg) 
00108 {
00109     // Raw data
00110     bb->pressure_raw_data.top_right = (msg[0] << 8) | msg[1];
00111     bb->pressure_raw_data.bottom_right = (msg[2] << 8) | msg[3];
00112     bb->pressure_raw_data.top_left = (msg[4] << 8) | msg[5];
00113     bb->pressure_raw_data.bottom_left = (msg[6] << 8) | msg[7];
00114     
00115     /* Weight in kg (we must interpole here) */
00116 
00117     // Top Right
00118     if(bb->pressure_raw_data.top_right <= bb->cal_medium_weight.top_right) {
00119         bb->pressure_weight.top_right = 17.0 * (bb->pressure_raw_data.top_right - bb->cal_low_weight.top_right) / (float)(bb->cal_medium_weight.top_right - bb->cal_low_weight.top_right);
00120     }
00121     else if(bb->pressure_raw_data.top_right <= bb->cal_high_weight.top_right) {
00122         bb->pressure_weight.top_right = 17.0 * (bb->pressure_raw_data.top_right - bb->cal_medium_weight.top_right) / (float)(bb->cal_high_weight.top_right - bb->cal_medium_weight.top_right) + 17.0;   
00123     }
00124     else {
00125         bb->pressure_weight.top_right = 17.0 * (bb->pressure_raw_data.top_right - bb->cal_high_weight.top_right) / (float)(bb->cal_high_weight.top_right - bb->cal_medium_weight.top_right) + 34.0;         
00126     }
00127     
00128     // Bottom Right
00129     if(bb->pressure_raw_data.bottom_right <= bb->cal_medium_weight.bottom_right) {
00130         bb->pressure_weight.bottom_right = 17.0 * (bb->pressure_raw_data.bottom_right - bb->cal_low_weight.bottom_right) / (float)(bb->cal_medium_weight.bottom_right - bb->cal_low_weight.bottom_right);
00131     }

00134     }
00135     else {
00136         bb->pressure_weight.bottom_right = 17.0 * (bb->pressure_raw_data.bottom_right - bb->cal_high_weight.bottom_right) / (float)(bb->cal_high_weight.bottom_right - bb->cal_medium_weight.bottom_right) + 34.0;          
00137     }
00138     
00139     // Top Left
00140     if(bb->pressure_raw_data.top_left <= bb->cal_medium_weight.top_left) {
00141         bb->pressure_weight.top_left = 17.0 * (bb->pressure_raw_data.top_left - bb->cal_low_weight.top_left) / (float)(bb->cal_medium_weight.top_left - bb->cal_low_weight.top_left);
00142     }
00143     else if(bb->pressure_raw_data.top_left <= bb->cal_high_weight.top_left) {

00145     }
00146     else {
00147         bb->pressure_weight.top_left = 17.0 * (bb->pressure_raw_data.top_left - bb->cal_high_weight.top_left) / (float)(bb->cal_high_weight.top_left - bb->cal_medium_weight.top_left) + 34.0;          
00148     }
00149     
00150     // Botton Left
00151     if(bb->pressure_raw_data.bottom_left <= bb->cal_medium_weight.bottom_left) {
00152         bb->pressure_weight.bottom_left = 17.0 * (bb->pressure_raw_data.bottom_left - bb->cal_low_weight.bottom_left) / (float)(bb->cal_medium_weight.bottom_left - bb->cal_low_weight.bottom_left);
00153     }
00154     else if(bb->pressure_raw_data.bottom_left <= bb->cal_high_weight.bottom_left) {
00155         bb->pressure_weight.bottom_left = 17.0 * (bb->pressure_raw_data.bottom_left - bb->cal_medium_weight.bottom_left) / (float)(bb->cal_high_weight.bottom_left - bb->cal_medium_weight.bottom_left) + 17.0; 
00156     }
00157     else {
00158         bb->pressure_weight.bottom_left = 17.0 * (bb->pressure_raw_data.bottom_left - bb->cal_high_weight.bottom_left) / (float)(bb->cal_high_weight.bottom_left - bb->cal_medium_weight.bottom_left) + 34.0;           
00159     }
00160     
00161     bb->pressure_weight.weight = bb->pressure_weight.top_right + bb->pressure_weight.top_left + bb->pressure_weight.bottom_right + bb->pressure_weight.bottom_left;
