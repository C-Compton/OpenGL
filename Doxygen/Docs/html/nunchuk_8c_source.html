00167 static void nunchuk_pressed_buttons(struct nunchuk_t* nc, byte now) {
00168     /* message is inverted (0 is active, 1 is inactive) */
00169     now = ~now & NUNCHUK_BUTTON_ALL;
00170 
00171     /* pressed now & were pressed, then held */
00172     nc->btns_held = (now & nc->btns);
00173 
00174     /* were pressed or were held & not pressed now, then released */
00175     nc->btns_released = ((nc->btns | nc->btns_held) & ~now);
00176 
00177     /* buttons pressed now */
00178     nc->btns = now;
00179 }
