00089 void motion_plus_event(struct motion_plus_t* mp, byte* msg) 
00090 {
00091     // Check if the gyroscope is in fast or slow mode (0 if rotating fast, 1 if slow or still)
00092     mp->acc_mode = ((msg[4] & 0x2) << 1) | ((msg[3] & 0x1) << 1) | ((msg[3] & 0x2) >> 1); 
00093     
00094     mp->a_raw_gyro.roll = ((msg[4] & 0xFC) << 6) | msg[1];
00095     mp->a_raw_gyro.pitch = ((msg[5] & 0xFC) << 6) | msg[2];
00096     mp->a_raw_gyro.yaw = ((msg[3] & 0xFC) << 6) | msg[0];
00097 
00098     // First calibration
00099     if((mp->a_raw_gyro.roll > 5000) && (mp->a_raw_gyro.pitch > 5000) && (mp->a_raw_gyro.yaw > 5000) && 
00100         !(mp->cal_gyro.roll) && !(mp->cal_gyro.pitch) && !(mp->cal_gyro.yaw)) {
00101         wiic_calibrate_motion_plus(mp); 
00102     }
00103     
00104     if(mp->smooth)
00105         motion_plus_apply_smoothing(mp);
00106     else {
00107         mp->raw_gyro.roll = mp->a_raw_gyro.roll;
00108         mp->raw_gyro.pitch = mp->a_raw_gyro.pitch;
00109         mp->raw_gyro.yaw = mp->a_raw_gyro.yaw;
00110     }
00111                 
00112     // Calculate angular rates in deg/sec and performs some simple filtering (both unsmoothed and smoothed)
00113     calculate_gyro_rates(mp,&(mp->a_raw_gyro),&(mp->a_gyro_rate));
00114     calculate_gyro_rates(mp,&(mp->raw_gyro),&(mp->gyro_rate));
00115 }
