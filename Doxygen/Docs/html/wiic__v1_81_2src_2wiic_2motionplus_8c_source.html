52 void calculate_gyro_rates(struct motion_plus_t* mp)
53 {
54  short int tmp_r, tmp_p, tmp_y;
55  float tmp_roll, tmp_pitch, tmp_yaw;
56 
57  // We consider calibration data
58  tmp_r = mp->raw_gyro.r - mp->cal_gyro.r;
59  tmp_p = mp->raw_gyro.p - mp->cal_gyro.p;
60  tmp_y = mp->raw_gyro.y - mp->cal_gyro.y;
61 
62  // We convert to degree/sec according to fast/slow mode
63  if(mp->acc_mode & 0x04)
64  tmp_roll = tmp_r / 20.0;
65  else
66  tmp_roll = tmp_r / 4.0;
67 
68  if(mp->acc_mode & 0x02)
69  tmp_pitch = tmp_p / 20.0;
70  else
71  tmp_pitch = tmp_p / 4.0;
72 
73  if(mp->acc_mode & 0x01)
74  tmp_yaw = tmp_y / 20.0;
75  else
76  tmp_yaw = tmp_y / 4.0;
77 
78  // Simple filtering
79  if(fabs(tmp_roll) < 0.5)
80  tmp_roll = 0.0;
81  if(fabs(tmp_pitch) < 0.5)
82  tmp_pitch = 0.0;
83  if(fabs(tmp_yaw) < 0.5)
84  tmp_yaw = 0.0;
85 
86  mp->angle_rate_gyro.r = tmp_roll;
87  mp->angle_rate_gyro.p = tmp_pitch;
88  mp->angle_rate_gyro.y = tmp_yaw;
89 }

97 void motion_plus_event(struct motion_plus_t* mp, byte* msg)
98 {
99  // Check if the gyroscope is in fast or slow mode (0 if rotating fast, 1 if slow or still)
100  mp->acc_mode = ((msg[4] & 0x2) << 1) | ((msg[3] & 0x1) << 1) | ((msg[3] & 0x2) >> 1);
101 
102  mp->raw_gyro.r = ((msg[4] & 0xFC) << 6) | msg[1];
103  mp->raw_gyro.p = ((msg[5] & 0xFC) << 6) | msg[2];
104  mp->raw_gyro.y = ((msg[3] & 0xFC) << 6) | msg[0];
105 
106  // First calibration
107  if((mp->raw_gyro.r > 5000) && (mp->raw_gyro.p > 5000) && (mp->raw_gyro.y > 5000) &&
108  !(mp->cal_gyro.r) && !(mp->cal_gyro.p) && !(mp->cal_gyro.y)) {
109  wiic_calibrate_motion_plus(mp);
110  }
111 
112  // Calculate angular rates in deg/sec and performs some simple filtering
113  calculate_gyro_rates(mp);
114 }

125 int motion_plus_handshake(struct wiimote_t* wm, byte* data, unsigned short len)
126 {
127  WIIMOTE_DISABLE_STATE(wm, WIIMOTE_STATE_EXP);
128  WIIMOTE_DISABLE_STATE(wm, WIIMOTE_STATE_EXP_FAILED);
129  WIIMOTE_DISABLE_STATE(wm, WIIMOTE_STATE_EXP_HANDSHAKE);
130 
131  unsigned int val = (data[2] << 24) | (data[3] << 16) | (data[4] << 8) | data[5];
132 
133  if(val == EXP_ID_CODE_MOTION_PLUS) {
134  /* handshake done */
135  wm->event = WIIC_MOTION_PLUS_INSERTED;
136  wm->exp.type = EXP_MOTION_PLUS;
137 
138  WIIMOTE_ENABLE_STATE(wm,WIIMOTE_STATE_EXP);
139  WIIMOTE_ENABLE_STATE(wm,WIIMOTE_STATE_MOTION_PLUS);
140 
141  // Init gyroscopes
142  wm->exp.mp.cal_gyro.r = 0;
143  wm->exp.mp.cal_gyro.p = 0;
144  wm->exp.mp.cal_gyro.y = 0;
145  wm->exp.mp.orient.roll = 0.0;
146  wm->exp.mp.orient.pitch = 0.0;
147  wm->exp.mp.orient.yaw = 0.0;
148  wm->exp.mp.raw_gyro_threshold = 10;
149  }
150  else {
151  WIIC_ERROR("Unable to activate Motion Plus");
152  wiic_set_motion_plus(wm,0);
153  return 0;
154  }
155 
156  return 1;
157 }

165 void wiic_set_motion_plus(struct wiimote_t *wm, int status)
166 {
167  byte val;
168  byte* buf = 0;
169  byte* tmp = 0;
170 
171  // If we're handshaking other expansions, than skip this
172  if(WIIMOTE_IS_SET(wm,WIIMOTE_STATE_EXP_HANDSHAKE))
173  return;
174 
175  WIIMOTE_ENABLE_STATE(wm, WIIMOTE_STATE_EXP_HANDSHAKE);
176  if(status) {
177  // We initialize the motion plus
178  val = 0x55;
179  wiic_write_data(wm,WM_MOTION_PLUS_INIT,&val,1);
180  usleep(10000);
181 
182  // We initialize the motion plus
183  val = 0x04;
184  wiic_write_data(wm,WM_MOTION_PLUS_ENABLE,&val,1);
185  usleep(10000);
186 
187  // Callback to check if the init process performed right
188  tmp = (byte*)malloc(sizeof(byte)*6);
189  wiic_read_data_cb(wm, motion_plus_handshake, tmp, WM_MOTION_PLUS_ID_ADDR, 6);
190  }
191  else {
192  disable_expansion(wm);
193  WIIMOTE_DISABLE_STATE(wm,WIIMOTE_STATE_MOTION_PLUS);
194  val = 0x55;
195  wiic_write_data(wm,WM_MOTION_PLUS_DISABLE,&val,1);
196  WIIMOTE_DISABLE_STATE(wm, WIIMOTE_STATE_EXP_HANDSHAKE);
197  }
198 }
