00173 static void classic_ctrl_pressed_buttons(struct classic_ctrl_t* cc, short now) {
00174     /* message is inverted (0 is active, 1 is inactive) */
00175     now = ~now & CLASSIC_CTRL_BUTTON_ALL;
00176 
00177     /* pressed now & were pressed, then held */
00178     cc->btns_held = (now & cc->btns);
00179 
00180     /* were pressed or were held & not pressed now, then released */
00181     cc->btns_released = ((cc->btns | cc->btns_held) & ~now);
00182 
00183     /* buttons pressed now */
00184     cc->btns = now;
