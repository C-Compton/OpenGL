00085         /*
00086          *  MACOSX

00114 
00115         if (!wm) return 0;

00118         tv.tv_sec = 0;
00119         tv.tv_usec = 500;
00120 
00121         FD_ZERO(&fds);
00122 
00123         for (i = 0; i < wiimotes; ++i) {
00124             /* only poll it if it is connected */

00150                 continue;
00151 
00152             if (FD_ISSET(wm[i]->in_sock, &fds)) {
00153                 /* clear out the event buffer */

00171                     }
00172 
00173                     continue;
00174                 }
00175                 if (!r) {
00176                     /* remote disconnect */
00177                     wiic_disconnected(wm[i]);

00179                     continue;
00180                 }

00183                 propagate_event(wm[i], wm[i]->event_buf[1], wm[i]->event_buf+2);
00184                 evnt += (wm[i]->event != WIIC_NONE);
00185             } else {

00188         }

00200 static void idle_cycle(struct wiimote_t* wm) {
00201     /*
00202      *  Smooth the angles.
00203      *
00204      *  This is done to make sure that on every cycle the orientation
00205      *  angles are smoothed.  Normally when an event occurs the angles
00206      *  are updated and smoothed, but if no packet comes in then the
00207      *  angles remain the same.  This means the angle WiiC reports
00208      *  is still an old value.  Smoothing needs to be applied in this
00209      *  case in order for the angle it reports to converge to the true
00210      *  angle of the device.
00211      */
00212     if (WIIC_USING_ACC(wm) && WIIMOTE_IS_FLAG_SET(wm, WIIC_SMOOTHING)) {
00213         apply_smoothing(&wm->gforce, wm->accel_calib.st_alpha);
00214     }
00215 
00216     /* clear out any old read requests */
00217     clear_dirty_reads(wm);
00218 }

00387 void wiic_pressed_buttons(struct wiimote_t* wm, byte* msg) {
00388     short now;
00389 
00390     /* convert to big endian */
00391     now = BIG_ENDIAN_SHORT(*(short*)msg) & WIIMOTE_BUTTON_ALL;
00392 
00393     /* pressed now & were pressed, then held */
00394     /* 
00395      * FIXME - With motion sensing disabled and no other activities
00396      * it's impossible to state if a button is held, since no other 
00397      * report will be sent.
00398      */
00399     wm->btns_held = (now & wm->btns);
00400     
00401     /* were pressed or were held & not pressed now, then released */
00402     wm->btns_released = ((wm->btns | wm->btns_held) & ~now);
00403 
00404     /* buttons pressed now */
00405     wm->btns = now;
00406 }

00421 static void event_data_read(struct wiimote_t* wm, byte* msg) {
00422     /* we must always assume the packet received is from the most recent request */
00423     byte err;
00424     byte len;
00425     unsigned short offset;
00426     struct read_req_t* req = wm->read_req;
00427 
00428     wiic_pressed_buttons(wm, msg);
00429 
00430     /* find the next non-dirty request */
00431     while (req && req->dirty)
00432         req = req->next;
00433 
00434     /* if we don't have a request out then we didn't ask for this packet */
00435     if (!req) {
00436         WIIC_WARNING("Received data packet when no request was made.");
00437         return;
00438     }
00439 
00440     err = msg[2] & 0x0F;
00441 
00442     if (err == 0x08)
00443         WIIC_WARNING("Unable to read data - address does not exist.");
00444     else if (err == 0x07)
00445         WIIC_WARNING("Unable to read data - address is for write-only registers.");
00446     else if (err)
00447         WIIC_WARNING("Unable to read data - unknown error code %x.", err);
00448 
00449     if (err) {
00450         /* this request errored out, so skip it and go to the next one */
00451 
00452         /* delete this request */
00453         wm->read_req = req->next;
00454         free(req);
00455 
00456         /* if another request exists send it to the wiimote */
00457         if (wm->read_req)
00458             wiic_send_next_pending_read_request(wm);
00459 
00460         return;
00461     }
00462 
00463     len = ((msg[2] & 0xF0) >> 4) + 1;
00464     offset = BIG_ENDIAN_SHORT(*(unsigned short*)(msg + 3));
00465     req->addr = (req->addr & 0xFFFF);
00466 
00467     req->wait -= len;
00468     if (req->wait >= req->size)
00469         /* this should never happen */
00470         req->wait = 0;
00471 
00472     WIIC_DEBUG("Received read packet:");
00473     WIIC_DEBUG("    Packet read offset:   %i bytes", offset);
00474     WIIC_DEBUG("    Request read offset:  %i bytes", req->addr);
00475     WIIC_DEBUG("    Read offset into buf: %i bytes", offset - req->addr);
00476     WIIC_DEBUG("    Read data size:       %i bytes", len);
00477     WIIC_DEBUG("    Still need:           %i bytes", req->wait);
00478 
00479     /* reconstruct this part of the data */
00480     memcpy((req->buf + offset - req->addr), (msg + 5), len);
00481 
00482     #ifdef WITH_WIIC_DEBUG
00483     {
00484         int i = 0;
00485         printf("Read: ");
00486         for (; i < req->size - req->wait; ++i)
00487             printf("%x ", req->buf[i]);
00488         printf("\n");
00489     }
00490     #endif
00491 
00492     /* if all data has been received, execute the read event callback or generate event */
00493     if (!req->wait) {
00494         if (req->cb) {
00495             /* this was a callback, so invoke it now */
00496             req->cb(wm, req->buf, req->size);
00497 
00498             /* delete this request */
00499             wm->read_req = req->next;
00500             free(req);
00501         } else {
00502             /*
00503              *  This should generate an event.
00504              *  We need to leave the event in the array so the client
00505              *  can access it still.  We'll flag is as being 'dirty'
00506              *  and give the client one cycle to use it.  Next event
00507              *  we will remove it from the list.
00508              */
00509             wm->event = WIIC_READ_DATA;
00510             req->dirty = 1;
00511         }
00512 
00513         /* if another request exists send it to the wiimote */
00514         if (wm->read_req)
00515             wiic_send_next_pending_read_request(wm);
00516     }
00517 }

00531     int ir = 0;

00535      *  An event occured.

00537      *  event type during a handshake or expansion removal.
00538      */

00540 
00541     wiic_pressed_buttons(wm, msg);
00542

00544     if (msg[2] & WM_CTRL_STATUS_BYTE1_LED_1)    led[0] = 1;
00545     if (msg[2] & WM_CTRL_STATUS_BYTE1_LED_2)    led[1] = 1;

00547     if (msg[2] & WM_CTRL_STATUS_BYTE1_LED_4)    led[3] = 1;
00548

00550     if ((msg[2] & WM_CTRL_STATUS_BYTE1_ATTACHMENT) == WM_CTRL_STATUS_BYTE1_ATTACHMENT)
00551         attachment = 1;
00552 
00553     /* is the speaker enabled? */
00554     if ((msg[2] & WM_CTRL_STATUS_BYTE1_SPEAKER_ENABLED) == WM_CTRL_STATUS_BYTE1_SPEAKER_ENABLED)
00555         WIIMOTE_ENABLE_STATE(wm, WIIMOTE_STATE_SPEAKER);
00556 
00557     /* is IR sensing enabled? */
00558     if ((msg[2] & WM_CTRL_STATUS_BYTE1_IR_ENABLED) == WM_CTRL_STATUS_BYTE1_IR_ENABLED)
00559         ir = 1;
00560 
00561     /* find the battery level and normalize between 0 and 1 */
00562     wm->battery_level = (msg[5] / (float)WM_MAX_BATTERY_CODE);
00563 
00564     /* expansion port */
00565     if (attachment && !WIIMOTE_IS_SET(wm, WIIMOTE_STATE_EXP)) {
00566         /* send the initialization code for the attachment */
00567         handshake_expansion(wm, NULL, 0);

00580     if (exp_changed && WIIMOTE_IS_SET(wm, WIIMOTE_STATE_IR)) {

00582          *  Since the expansion status changed IR needs to
00583          *  be reset for the new IR report mode.
00584          */
00585         WIIMOTE_DISABLE_STATE(wm, WIIMOTE_STATE_IR);
00586         wiic_set_ir(wm, 1);
