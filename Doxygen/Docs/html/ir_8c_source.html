00544 static void fix_rotated_ir_dots(struct ir_dot_t* dot, float ang) {
00545     float s, c;
00546     int x, y;
00547     int i;
00548 
00549     if (!ang) {
00550         for (i = 0; i < 4; ++i) {
00551             dot[i].x = dot[i].rx;
00552             dot[i].y = dot[i].ry;
00553         }
00554         return;
00555     }
00556 
00557     s = sin(DEGREE_TO_RAD(ang));
00558     c = cos(DEGREE_TO_RAD(ang));
00559 
00560     /*
00561      *  [ cos(theta)  -sin(theta) ][ ir->rx ]
00562      *  [ sin(theta)  cos(theta)  ][ ir->ry ]
00563      */
00564 
00565     for (i = 0; i < 4; ++i) {
00566         if (!dot[i].visible)
00567             continue;
00568 
00569         x = dot[i].rx - (1024/2);
00570         y = dot[i].ry - (768/2);
00571 
00572         dot[i].x = (c * x) + (-s * y);
00573         dot[i].y = (s * x) + (c * y);
00574 
00575         dot[i].x += (1024/2);
00576         dot[i].y += (768/2);
00577     }
00578 }
