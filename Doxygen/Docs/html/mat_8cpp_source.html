00623   mat4 Ortho( const GLfloat left, const GLfloat right,
00624              const GLfloat bottom, const GLfloat top,
00625              const GLfloat zNear, const GLfloat zFar ) {
00626     mat4 c;
00627     c[0][0] = 2.0/(right - left);
00628     c[1][1] = 2.0/(top - bottom);
00629     c[2][2] = 2.0/(zNear - zFar);
00630     c[3][3] = 1.0;
00631     c[0][3] = -(right + left)/(right - left);
00632     c[1][3] = -(top + bottom)/(top - bottom);
00633     c[2][3] = -(zFar + zNear)/(zFar - zNear);
00634     return c;
00635   }

00656   mat4 Perspective( const GLfloat fovy, const GLfloat aspect,
00657                const GLfloat zNear, const GLfloat zFar) {
00658     GLfloat top = tan(fovy*DegreesToRadians/2) * zNear;
00659     GLfloat right = top * aspect;
00660

00662     c[0][0] = zNear/right;
00663     c[1][1] = zNear/top;
00664     c[2][2] = -(zFar + zNear)/(zFar - zNear);
00665     c[2][3] = -2.0*zFar*zNear/(zFar - zNear);
