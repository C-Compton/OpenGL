00063 void calculate_orientation(struct vec3f_t* in, struct ang3f_t* out) {
00064     float x, y, z;
00065 
00066     /*
00067      *  roll    - use atan(z / x)       [ ranges from -180 to 180 ]
00068      *  pitch   - use atan(z / y)       [ ranges from -180 to 180 ]
00069      *  yaw     - impossible to tell without IR
00070      */
00071 
00072     /* yaw - set to 0, IR will take care of it if it's enabled */
00073     out->yaw = 0.0f;
00074 
00075     /* find out how much it actually moved and normalize to +/- 1g */
00076     x = in->x;
00077     y = in->y;
00078     z = in->z;
00079 
00080     /* make sure x,y,z are between -1 and 1 for the tan functions */
00081     if (x < -1.0f)          x = -1.0f;
00082     else if (x > 1.0f)      x = 1.0f;
00083     if (y < -1.0f)          y = -1.0f;
00084     else if (y > 1.0f)      y = 1.0f;
00085     if (z < -1.0f)          z = -1.0f;
00086     else if (z > 1.0f)      z = 1.0f;
00087 
00088     /* if it is over 1g then it is probably accelerating and the gravity vector cannot be identified */
00089     if (abs(in->x) <= 1.0) {
00090         /* roll */
00091         x = -RAD_TO_DEGREE(atan2f(x, z));
00092 
00093         out->roll = x;
00094     }
00095 
00096     if (abs(in->y) <= 1.0) {
00097         /* pitch */
00098         y = RAD_TO_DEGREE(atan2f(y, z));
00099 
00100         out->pitch = y;
00101     }
00102 }

00143 void calc_joystick_state(struct joystick_t* js, float x, float y) {
00144     float rx, ry, ang;
00145 
00146     /*
00147      *  Since the joystick center may not be exactly:
00148      *      (min + max) / 2
00149      *  Then the range from the min to the center and the center to the max
00150      *  may be different.
00151      *  Because of this, depending on if the current x or y value is greater
00152      *  or less than the assoicated axis center value, it needs to be interpolated
00153      *  between the center and the minimum or maxmimum rather than between
00154      *  the minimum and maximum.
00155      *
00156      *  So we have something like this:
00157      *      (x min) [-1] ---------*------ [0] (x center) [0] -------- [1] (x max)
00158      *  Where the * is the current x value.
00159      *  The range is therefore -1 to 1, 0 being the exact center rather than
00160      *  the middle of min and max.
00161      */
00162     if (x == js->center.x)
00163         rx = 0;
00164     else if (x >= js->center.x)
00165         rx = ((float)(x - js->center.x) / (float)(js->max.x - js->center.x));
00166     else
00167         rx = ((float)(x - js->min.x) / (float)(js->center.x - js->min.x)) - 1.0f;
00168 
00169     if (y == js->center.y)
00170         ry = 0;
00171     else if (y >= js->center.y)
00172         ry = ((float)(y - js->center.y) / (float)(js->max.y - js->center.y));
00173     else
00174         ry = ((float)(y - js->min.y) / (float)(js->center.y - js->min.y)) - 1.0f;
00175 
00176     /* calculate the joystick angle and magnitude */
00177     ang = RAD_TO_DEGREE(atanf(ry / rx));
00178     ang -= 90.0f;
00179     if (rx < 0.0f)
00180         ang -= 180.0f;
00181     js->ang = absf(ang);
00182     js->mag = (float) sqrt((rx * rx) + (ry * ry));
00183 }
