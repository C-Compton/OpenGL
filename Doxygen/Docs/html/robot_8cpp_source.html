41 void acc2robot(CWiimote &wm)
42 {
43  bool enable = wm.Buttons.isPressed(CButtons::BUTTON_B);
44 
45  if(enable) {
46  // Acquire smoothed gravity vector
47  wm.Accelerometer.GetGravityVector(x, y, z);
48 
49  // Acquire orientation
50  wm.Accelerometer.GetOrientation(pitch, roll, yaw);
51 
52  // JOG
53  if(fabs(x) <= 0.25) // Still
54  robotJog = 0.0;
55  else
56  robotJog = MAX_JOG*roll*2.0/M_PI;
57 
58  robotJog > MAX_JOG ? robotJog = MAX_JOG : robotJog;
59  robotJog < -MAX_JOG ? robotJog = -MAX_JOG : robotJog;
60 
61  // SPEED
62  if(fabs(y) <= 0.25) // Still
63  robotSpeed = 0.0;
64  else
65  robotSpeed = MAX_SPEED*pitch*2.0/M_PI;
66 
67  robotSpeed > MAX_SPEED ? robotSpeed = MAX_SPEED : robotSpeed;
68  robotSpeed < -MAX_SPEED ? robotSpeed = -MAX_SPEED : robotSpeed;
69  }
70  else {
71  pitch = roll = yaw = x = y = z = 0.0;
72  robotSpeed = robotJog = 0.0;
73  }
74 }

76 int main(int argc, char** argv)
77 {
78  CWii wii; // Defaults to 4 remotes
79  std::vector<CWiimote>::iterator i;
80  int reloadWiimotes = 0;
81  int numFound;
82  int index;
83  bool help = false;
84  roll = pitch = yaw = x = y = z = robotSpeed = robotJog = 0.0; // Init
85 
86  string option;
87  if(argc == 2) {
88  option = string(argv[1]);
89  if(option == "help") {
90  cout << endl << "wiic-robot is a simple teleoperation interface to control a robot with a Wiimote." << endl;
91  cout << "wiic-robot is part of WiiC (http://wiic.sf.net)" << endl << endl;
92  cout << "Usage: wiic-robot" << endl;
93 
94  return 1;
95  }
96  }
97 
98  // Find and connect to the wiimotes
99  std::vector<CWiimote>& wiimotes = wii.FindAndConnect(1);
100 
101  if(!wiimotes.size()) {
102  cout << "Error: no connected Wiimote" << endl;
103  return 1;
104  }
105  CWiimote& wiimote = wiimotes[0];
106 
107  do
108  {
109  if(reloadWiimotes)
110  {
111  // Regenerate the list of wiimotes
112  wiimotes = wii.GetWiimotes();
113  reloadWiimotes = 0;
114  }
115 
116  //Poll the wiimotes to get the status like pitch or roll
117  if(wii.Poll())
118  {
119  for(i = wiimotes.begin(); i != wiimotes.end(); ++i)
120  {
121  // Use a reference to make working with the iterator handy.
122  CWiimote & wiimote = *i;
123  switch(wiimote.GetEvent())
124  {
125 
126  case CWiimote::EVENT_EVENT:
127  acc2robot(wiimote);
128  break;
129 
130  case CWiimote::EVENT_DISCONNECT:
131  case CWiimote::EVENT_UNEXPECTED_DISCONNECT:
132  reloadWiimotes = 1;
133  break;
134 
135  default:
136  break;
137  }
138  }
139  }
140 
141  } while(wiimotes.size()); // Go so long as there are wiimotes left to poll
142 
143  return 0;
144 }
