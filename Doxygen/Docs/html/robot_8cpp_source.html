00041 void acc2robot(CWiimote &wm)
00042 {
00043     bool enable = wm.Buttons.isPressed(CButtons::BUTTON_B);
00044     
00045     if(enable) {
00046         // Acquire smoothed gravity vector
00047         wm.Accelerometer.GetGravityVector(x, y, z);
00048     
00049         // Acquire orientation
00050         wm.Accelerometer.GetOrientation(pitch, roll, yaw);
00051         
00052         // JOG
00053         if(fabs(x) <= 0.25)  // Still
00054             robotJog = 0.0; 
00055         else 
00056             robotJog = MAX_JOG*roll*2.0/M_PI;
00057 
00058         robotJog > MAX_JOG ? robotJog = MAX_JOG : robotJog;
00059         robotJog < -MAX_JOG ? robotJog = -MAX_JOG : robotJog;
00060 
00061         // SPEED
00062         if(fabs(y) <= 0.25)  // Still
00063             robotSpeed = 0.0;   
00064         else 
00065             robotSpeed = MAX_SPEED*pitch*2.0/M_PI;
00066 
00067         robotSpeed > MAX_SPEED ? robotSpeed = MAX_SPEED : robotSpeed;
00068         robotSpeed < -MAX_SPEED ? robotSpeed = -MAX_SPEED : robotSpeed; 
00069     }
00070     else {
00071         pitch = roll = yaw = x = y = z = 0.0;
00072         robotSpeed = robotJog = 0.0;
00073     }
00074 }
