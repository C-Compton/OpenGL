00060     int i;
00061     int offset = 0;
00062 
00063     /*
00064      *  The good fellows that made the Guitar Hero 3 controller
00065      *  failed to factory calibrate the devices.  There is no
00066      *  calibration data on the device.
00067      */
00068 
00069     gh3->btns = 0;
00070     gh3->btns_held = 0;
00071     gh3->btns_released = 0;

00075     for (i = 0; i < len; ++i)
00076         data[i] = (data[i] ^ 0x17) + 0x17;
00077 
00078     if (data[offset] == 0xFF) {

00080          *  Sometimes the data returned here is not correct.
00081          *  This might happen because the wiimote is lagging
00082          *  behind our initialization sequence.
00083          *  To fix this just request the handshake again.
00084          *
00085          *  Other times it's just the first 16 bytes are 0xFF,
00086          *  but since the next 16 bytes are the same, just use
00087          *  those.
00088          */
00089         if (data[offset + 16] == 0xFF) {
00090             /* get the calibration data */
00091             byte* handshake_buf = malloc(EXP_HANDSHAKE_LEN * sizeof(byte));
00092 
00093             WIIC_DEBUG("Guitar Hero 3 handshake appears invalid, trying again.");
00094             wiic_read_data_cb(wm, handshake_expansion, handshake_buf, WM_EXP_MEM_CALIBR, EXP_HANDSHAKE_LEN);
00095 
00096             return 0;
00097         } else
00098             offset += 16;
00099     }
00100 
00101     /* joystick stuff */
00102     gh3->js.max.x = GUITAR_HERO_3_JS_MAX_X;
00103     gh3->js.min.x = GUITAR_HERO_3_JS_MIN_X;
00104     gh3->js.center.x = GUITAR_HERO_3_JS_CENTER_X;
00105     gh3->js.max.y = GUITAR_HERO_3_JS_MAX_Y;
00106     gh3->js.min.y = GUITAR_HERO_3_JS_MIN_Y;
00107     gh3->js.center.y = GUITAR_HERO_3_JS_CENTER_Y;

00156 static void guitar_hero_3_pressed_buttons(struct guitar_hero_3_t* gh3, short now) {
00157     /* message is inverted (0 is active, 1 is inactive) */
00158     now = ~now & GUITAR_HERO_3_BUTTON_ALL;
00159 
00160     /* pressed now & were pressed, then held */

00163     /* were pressed or were held & not pressed now, then released */

00165 
00166     /* buttons pressed now */
