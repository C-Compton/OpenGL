443 std::vector<CIRDot>& CIR::GetDots()
444 {
445  int index;
446 
447  // Empty the array of irdots before reloading
448  mpIRDotsVector.clear();
449 
450  for(index = 0 ; index < 4 ; index++)
451  {
452  CIRDot dot((struct ir_dot_t *) (&(mpWiimotePtr->ir.dot[index])));
453  if(dot.isVisible())

455  }
456 
457  return mpIRDotsVector;
458 }

954 std::vector<CWiimote>& CWii::FindAndConnect(int timeout, bool rumbleAck, bool autoreconnect)
955 {

958  int index;

970  cout << "Connected to " << (unsigned int)mpWiimotesVector.size() << " wiimotes" << endl;
971

974  {
975  // Use a reference to make working with the iterator handy.
976  CWiimote & wiimote = *i;
977 
978  //Rumble for 0.2 seconds as a connection ack
