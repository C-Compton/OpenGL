00139   // (By extension: drawMaxRows = (numRows-drawMinRows))

00141     ((int)(numCameras - (numRows * numMaxCols)) / (int)(numMinCols - numMaxCols)) :
00142     0;

00145   /* This is confusing as hell. Good luck! */

00147        (row < numRows) || (it != camList.end()); //Terminate on either cond. Just in case.
