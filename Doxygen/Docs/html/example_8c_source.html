62 // printf("\n\n--- EVENT [id %i] ---\n", wm->unid);

64  /* if a button is pressed, report it */
65  if (IS_PRESSED(wm, WIIMOTE_BUTTON_A)) printf("A pressed\n");
66  if (IS_PRESSED(wm, WIIMOTE_BUTTON_B)) printf("B pressed\n");
67  if (IS_PRESSED(wm, WIIMOTE_BUTTON_UP)) printf("UP pressed\n");
68  if (IS_PRESSED(wm, WIIMOTE_BUTTON_DOWN)) printf("DOWN pressed\n");
69  if (IS_PRESSED(wm, WIIMOTE_BUTTON_LEFT)) printf("LEFT pressed\n");
70  if (IS_PRESSED(wm, WIIMOTE_BUTTON_RIGHT)) printf("RIGHT pressed\n");
71  if (IS_PRESSED(wm, WIIMOTE_BUTTON_MINUS)) printf("MINUS pressed\n");
72  if (IS_PRESSED(wm, WIIMOTE_BUTTON_PLUS)) printf("PLUS pressed\n");
73  if (IS_PRESSED(wm, WIIMOTE_BUTTON_ONE)) printf("ONE pressed\n");
74  if (IS_PRESSED(wm, WIIMOTE_BUTTON_TWO)) printf("TWO pressed\n");
75  if (IS_PRESSED(wm, WIIMOTE_BUTTON_HOME)) printf("HOME pressed\n");
76 
77  /*
78  * Pressing minus will tell the wiimote we are no longer interested in movement.
79  * This is useful because it saves battery power.
80  */
81  if (IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_MINUS))
82  wiic_motion_sensing(wm, 0);
83 
84  /*
85  * Pressing plus will tell the wiimote we are interested in movement.
86  */
87  if (IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_PLUS))
88  wiic_motion_sensing(wm, 1);
89 
90  if(IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_LEFT))
91  wiic_set_motion_plus(wm,0);
92  if(IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_RIGHT))
93  wiic_set_motion_plus(wm,1);
94  /*
95  * Pressing B will toggle the rumble
96  *
97  * if B is pressed but is not held, toggle the rumble
98  */
99  if (IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_B))
100  wiic_toggle_rumble(wm);
101 
102  if (IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_UP))
103  wiic_set_ir(wm, 1);
104  if (IS_JUST_PRESSED(wm, WIIMOTE_BUTTON_DOWN))
105  wiic_set_ir(wm, 0);
106 
107  /* if the accelerometer is turned on then print angles */
108  if (WIIC_USING_ACC(wm)) {
109  printf("wiimote roll = %f\n", wm->orient.angle.roll);
110  printf("wiimote pitch = %f\n", wm->orient.angle.pitch);
111  printf("wiimote yaw = %f\n", wm->orient.angle.yaw);
112  }
113 
114  /* if the Motion Plus is turned on then print angle rates */
115  if (WIIC_USING_MOTION_PLUS(wm)) {
116  printf("motion plus roll rate = %f [%f]\n", wm->exp.mp.gyro_rate.roll, wm->exp.mp.a_gyro_rate.roll);
117  printf("motion plus pitch rate = %f [%f]\n", wm->exp.mp.gyro_rate.pitch, wm->exp.mp.a_gyro_rate.pitch);
118  printf("motion plus yaw rate = %f [%f]\n", wm->exp.mp.gyro_rate.yaw, wm->exp.mp.a_gyro_rate.yaw);
119  }
120 
121  /*
122  * If IR tracking is enabled then print the coordinates
123  * on the virtual screen that the wiimote is pointing to.
124  *
125  * Also make sure that we see at least 1 dot.
126  */
127  if (WIIC_USING_IR(wm)) {
128  int i = 0;
129

135  }

138  printf("IR z distance: %f\n", wm->ir.z);
139  }

296 int main(int argc, char** argv) {
297  wiimote** wiimotes;
298  int found, connected;
299 
300  /*
301  * Initialize an array of wiimote objects.
302  *
303  * The parameter is the number of wiimotes I want to create.
304  */
305  wiimotes = wiic_init(MAX_WIIMOTES);
306 
307  /*
308  * Find wiimote devices
309  *
310  * Now we need to find some wiimotes.
311  * Give the function the wiimote array we created, and tell it there
312  * are MAX_WIIMOTES wiimotes we are interested in.
313  *
314  * Set the timeout to be 5 seconds.
315  *
316  * This will return the number of actual wiimotes that are in discovery mode.
317  */
318  found = wiic_find(wiimotes, MAX_WIIMOTES, 5);
319  if (!found) {
320  printf ("No wiimotes found.");
321  return 0;
322  }
323 
324  /*
325  * Connect to the wiimotes
326  *
327  * Now that we found some wiimotes, connect to them.
328  * Give the function the wiimote array and the number
329  * of wiimote devices we found.
330  *
331  * This will return the number of established connections to the found wiimotes.
332  */
333  connected = wiic_connect(wiimotes, found, 0);
334  if (connected)
335  printf("Connected to %i wiimotes (of %i found).\n", connected, found);
336  else {
337  printf("Failed to connect to any wiimote.\n");
338  return 0;
339  }
340 
341  /*
342  * Now set the LEDs and rumble for a second so it's easy
343  * to tell which wiimotes are connected (just like the wii does).
344  */
345  wiic_set_leds(wiimotes[0], WIIMOTE_LED_1);
346  wiic_rumble(wiimotes[0], 1);
347  usleep(200000);
348  wiic_rumble(wiimotes[0], 0);
349 
350  /*
351  * Maybe I'm interested in the battery power of the 0th
352  * wiimote. This should be WIIMOTE_ID_1 but to be sure
353  * you can get the wiimote assoicated with WIIMOTE_ID_1
354  * using the wiic_get_by_id() function.
355  *
356  * A status request will return other things too, like
357  * if any expansions are plugged into the wiimote or
358  * what LEDs are lit.
359  */
360  wiic_status(wiimotes[0]);
361 
362  printf("\nPress PLUS (MINUS) to enable (disable) Motion Sensing Report (only accelerometers)\n");
363  printf("Press RIGHT (LEFT) to enable (disable) Motion Plus (requires Motion Sensing enabled)\n");
364  printf("Press UP (DOWN) to enable (disable) IR camera (requires some IR led)\n");
365 
366  /*
367  * This is the main loop
368  *
369  * wiic_poll() needs to be called with the wiimote array
370  * and the number of wiimote structures in that array
371  * (it doesn't matter if some of those wiimotes are not used
372  * or are not connected).
373  *
374  * This function will set the event flag for each wiimote
375  * when the wiimote has things to report.
376  */
377  while (!exiting) {
378  if (wiic_poll(wiimotes, connected)) {
379  /*
380  * This happens if something happened on any wiimote.
381  * So go through each one and check if anything happened.
382  */
383  int i = 0;
384  for (; i < MAX_WIIMOTES; ++i) {
385  switch (wiimotes[i]->event) {
386  case WIIC_EVENT:
387  // a generic event occured
388  handle_event(wiimotes[i]);
389  break;
390 
391  case WIIC_STATUS:
392  // a status event occured
393  handle_ctrl_status(wiimotes[i]);
394  break;
395 
396  case WIIC_DISCONNECT:
397  case WIIC_UNEXPECTED_DISCONNECT:
398  // the wiimote disconnected
399  handle_disconnect(wiimotes[i]);
400  break;
401 
402  case WIIC_READ_DATA:
403  /*
404  * Data we requested to read was returned.
405  * Take a look at wiimotes[i]->read_req
406  * for the data.
407  */
408  break;
409 
410  case WIIC_NUNCHUK_INSERTED:
411  /*
412  * a nunchuk was inserted
413  * This is a good place to set any nunchuk specific
414  * threshold values. By default they are the same
415  * as the wiimote.
416  */
417  printf("Nunchuk inserted.\n");
418  break;
419 
420  case WIIC_CLASSIC_CTRL_INSERTED:
421  printf("Classic controller inserted.\n");
422  break;
423 
424  case WIIC_GUITAR_HERO_3_CTRL_INSERTED:
425  // some expansion was inserted
426  handle_ctrl_status(wiimotes[i]);
427  break;
428 
429  case WIIC_MOTION_PLUS_INSERTED:
430  printf("Motion Plus inserted.\n");
431  break;
432 
433  case WIIC_BALANCE_BOARD_INSERTED:
434  printf("Balance Board connected.\n");
435  break;
436 
437  case WIIC_BALANCE_BOARD_REMOVED:
438  printf("Balance Board disconnected.\n");
439  break;
440 
441  case WIIC_NUNCHUK_REMOVED:
442  case WIIC_CLASSIC_CTRL_REMOVED:
443  case WIIC_GUITAR_HERO_3_CTRL_REMOVED:
444  case WIIC_MOTION_PLUS_REMOVED:
445  // some expansion was removed
446  handle_ctrl_status(wiimotes[i]);
447  printf("An expansion was removed.\n");
448  break;
449 
450  default:
451  break;
452  }
453  }
454  }
455  }
456 
457  /*
458  * Disconnect the wiimotes
459  */
460  wiic_cleanup(wiimotes, MAX_WIIMOTES);
461 
462  return 0;
463 }
