00163 void WiiHandleBB( CWiimote &wm ) {
00164   enum bb_sensor { 
00165     TOP_LEFT,
00166     TOP_RIGHT,
00167     BOT_LEFT,
00168     BOT_RIGHT,
00169     TOT_WEIGHT
00170   };
00171 
00172   static bool tare = false;
00173   static float tare_val[5] = { 0, 0, 0, 0, 0 };
00174   static float raw_val[5] = { 0, 0, 0, 0, 0 };
00175   static float adj_val[5] = { 0, 0, 0, 0, 0 };
00176   static size_t tare_polls = 0;
00177 
00178   if(wm.Buttons.isPressed(CButtons::BUTTON_A)) {
00179     std::cerr << "Enabling Tare\n";
00180     tare = true;
00181     tare_polls = 0;
00182     for (size_t i = 0; i < 5; ++i ) tare_val[i] = 0;
00183     bb_magnitudes = vec3( 0, 0, 0 );
00184   }
00185   CBalanceBoard &bb = wm.ExpansionDevice.BalanceBoard;
00186   bb.WeightSensor.GetWeight( raw_val[TOT_WEIGHT],
00187                  raw_val[TOP_LEFT],
00188                  raw_val[TOP_RIGHT],
00189                  raw_val[BOT_LEFT],
00190                  raw_val[BOT_RIGHT] );
00191 
00192   /* Compute our adjusted values. */
00193   adj_val[TOT_WEIGHT] = 0;
00194   for ( size_t i = 0; i < 4; ++i ) {
00195     adj_val[i] = raw_val[i] - tare_val[i];
00196     adj_val[TOT_WEIGHT] += adj_val[i];
00197   }
00198   
00199   /* We are now either going to zero the scale, 
00200      or ignore the inputs if they are too small. */
00201   if (tare) {
00202     for ( size_t i = 0; i < 4; ++i ) tare_val[i] += raw_val[i];
00203     if (++tare_polls >= 1000) {
00204       std::cerr << "Tare complete.\n";
00205       tare_val[4] = 0;
00206       tare_polls = 0;
00207       tare = false;
00208       for ( size_t i = 0; i < 4; ++i ) {
00209     tare_val[i] = tare_val[i] / 1000;
00210     std::cerr << "Tare[" << i << "] == " << tare_val[i] << ".\n";
00211     tare_val[4] += tare_val[i];
00212       }
00213     }
00214     return; /* Return early: do not compute anything with weird half-tared values. */
00215   } else if (adj_val[TOT_WEIGHT] < 10) {
00216     bb_magnitudes = vec3( 0, 0, 0 );
00217     return;  
00218   }
00219 
00220   /* Compute X and Y magnitudes. */
00221   float surge_pct = (adj_val[TOP_LEFT] + adj_val[TOP_RIGHT]
00222              -adj_val[BOT_LEFT] - adj_val[BOT_RIGHT]) / adj_val[TOT_WEIGHT];
00223   float sway_pct = (adj_val[TOP_RIGHT] + adj_val[BOT_RIGHT]
00224             -adj_val[TOP_LEFT] - adj_val[BOT_LEFT]) / adj_val[TOT_WEIGHT];
00225 
00226   /* Even with zeroing the scale and ignoring small weights,
00227      Sometimes the sensors report something weird.
00228      With this we ensure that we've normalized to the [-1,1] range. */
00229   if (surge_pct < -1) surge_pct = -1;
00230   else if (surge_pct > 1) surge_pct = 1;
00231   if (sway_pct < -1) sway_pct = -1;
00232   else if (sway_pct > 1) sway_pct = 1;
00233 
00234   bb_magnitudes.y = sway_pct;
00235   bb_magnitudes.z = surge_pct;
00236 
00237   if (0) {
00238     printf( "Balance Board Raw Weights: {" );
00239     for ( size_t i = 0; i < 5; ++i ) printf( "%6f, ", raw_val[i] );
00240     printf( "}\n" );
00241     printf( "Balance Board Adj Weights: {" );
00242     for ( size_t i = 0; i < 5; ++i ) printf( "%6f, ", adj_val[i] );
00243     printf( "}\n" );  
00244     printf( "SURGE: %6f; SWAY: %6f\n", surge_pct, sway_pct );
00245   }
00246   
00247 }
