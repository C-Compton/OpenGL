00074   // Next, we load the real data in parts.  We need to specify the
00075   //   correct byte offset for placing the color data after the point

00079   glBufferSubData( GL_ARRAY_BUFFER, 0, sizeof(points), points );
00080   glBufferSubData( GL_ARRAY_BUFFER, sizeof(points), sizeof(colors), colors );

00082   // Load shaders and use the resulting shader program

00084   glUseProgram( program );
00085   
00086   // Initialize the vertex position attribute from the vertex shader  
00087   GLuint vPosition = glGetAttribLocation( program, "vPosition" );
00088   glEnableVertexAttribArray( vPosition );
00089   glVertexAttribPointer( vPosition, 4, GL_FLOAT, GL_FALSE, 0,
00090              BUFFER_OFFSET(0) );
00091   
00092   // Likewise, initialize the vertex color attribute.  Once again, we
00093   //    need to specify the starting offset (in bytes) for the color
00094   //    data.  Just like loading the array, we use "sizeof(points)"
00095   //    to determine the correct value.
00096   GLuint vColor = glGetAttribLocation( program, "vColor" );
00097   glEnableVertexAttribArray( vColor );
